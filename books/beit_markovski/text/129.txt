לָהּ , לָהּ בִּלְבַד מֵאֲשֶׁר סָבְתָא מוֹרִיָּה סָבְתָא הַמְּיֻחֶדֶת שֶׁלִּי ,
שֶׁלָּנוּ .
    קְטַנָּה וְרָזָה , צְנוּעָה וּשְׁקֵטָה , כָּל צַעֲדָהּ – כְּמִין הִתְנַצְּלוּת
וְהִצְטַנְּעוּת עַל מָקוֹם שֶׁל מִישֶׁהוּ שֶׁתָּפְסָה כִּבְיָכוֹל , בְּנוֹכְחוּתָהּ
וְלֹא הִיא . הִיא לֹא תָּפְסָה מָקוֹם שֶׁל אִישׁ כְּשֵׁם שֶׁלֹּא פָּגְעָה
בְּאִישׁ וְתָמִיד נִזְהֲרָה בִּכְבוֹדוֹ שֶׁל אָדָם . אַךְ נוֹכְחוּתָהּ לְגַביי -
עוֹלָם וּמְלוֹאוֹ .
    כַּאֲשֶׁר סָבְתָא הָיְתָה נָמָה מְנוּחַת הַצָּהֳרַיִם שֶׁקֶט הָיָה שׁוֹרֵר
סָבִיב צְרִיפָהּ , לְהַפְרִיעַ מְנוּחָתָהּ הָיָה דָּבָר שֶׁלֹּא יֵעָשֶׂה . וְהָאוֹת
לִי , לַיֶּלֶד , שֶׁאֶפְשָׁר לְשַׂחֵק בְּחָפְשִׁיּוּת - הָיָה הַתְּרִיס הַנִּפְתָּח
לְיַד הַמַּנְדָּרִינָה וְדְמוּתָהּ הַמּוֹפִיעָה בַּחַלּוֹן .
    לְהַשְׁכָּמָתָהּ הָיוּ לִי כַּמָּה אוֹתוֹת שֶׁהָרִאשׁוֹן בָּהֶם , צְעָדִים
לַמִּטְבָּח וְקוֹל מַיִם נִגָּרִים מִבֶּרֶז , צְעָדִים בַּמִּסְדְּרוֹן וְ... הַדֶּלֶת
נִפְתַּחַת וְסָבְתָא נִצֶּבֶת לְבוּשָׁה בִּקְפִידָה , תַּלְתַּלֶּיהָ סְרוּקִים
וּמְסֻדָּרִים כָּל אֶחָד בִּמְקוֹמוֹ וְסִכָּה שֶׁשְּׁלֹשָׁה עִגּוּלִים חָבְרוּ
בָּהּ בְּשׁוּרָה וְהָאֶמְצָעִי — אֶבֶן תְּכֵלָּה בּוֹ , פּוֹרֶפֶת שִׂמְלָתָהּ
וּמַשְׁלִימָה לְבוּשָׁהּ , סָבְתָא קָמָה . וּבְלֶכְתָּהּ בָּרְחוֹב מִשְׁתַּדֶּלֶת
לְהַקְדִּים שָׁלוֹם לְכָל אֶחָד וּמְשִׁיבָה בְּמֵאוֹר פָּנִים לִמְבָרְכֶיהָ .
סָבְתָא הוֹלֶכֶת .
    שְׁתֵּי מַזְכָּרוֹת נִשְׁאֲרוּ לִי מִסָּבְתָא : הָאַהֲבָה לִקְרִיאָה וְהַסֵּפֶר
הָרִאשׁוֹן שֶׁקִּבַּלְתִּי מִמֶּנָּה לְיוֹם הֻלַּדְתִּי הַשְּׁבִיעִי , "לוּבֶּנְגּוּלוּ
מֶלֶךְ זוּלוּ ", אוֹתוֹ קוֹרֵאת כָּעֵת בִּתִּי סִגָּלִית .

אֶחָד מִתַּעֲנוּגוֹתַי הַגְּדוֹלִים הָיָה לְהַקְדִּימָהּ וְלָלֶכֶת לַדֹּאַר
לִשְׁאוֹל שֶׁמָּא יֵשׁ חֲבִילַת סְפָרִים בַּעֲבוּרָהּ לְהֲבִיאָם אֵלֶיהָ ,
לִפְתֹּחַ יַחַד עִמָּהּ אֶת אֲרִיזַת הַנְּיָר החוּם , לִנְבֹּר בַּעֲרֵימַת
הַסְּפָרִים הַחֲדָשִׁים , לְמַיְּנָם יַחַד עִמָּהּ וּלְסִיּוּם – לְקַבֵּל מִמֶּנָּה
מוֹדָעָה , לְפִרְסוּם לְיַד הַצַּרְכָּנִיָּה , בָּהּ הִיא מוֹדִיעָה עַל הַסְּפָרִים
