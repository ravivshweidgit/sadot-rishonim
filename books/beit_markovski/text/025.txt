הַשַּׁיִט עַל הַכִּנֶּרֶת הָיָה מַקְסִים, הַיָּם - כְּחַלְחַל, חָלָק
וּמַבְרִיק כִּרְאִי וְהַמַּיִם זַכִּים וּצְלוּלִים.
- "תִּשְׁתּוּ מֵהַמַּיִם, הֵם טוֹבִים וּמְתוּקִים" אָמְרוּ הַשַּׁיָּטִים.
שָׁאַבְנוּ מְלוֹא הַסִּפְּלוֹנִים וְשָׁתִינוּ.
- "נָכוֹן, מְתוּקִים", נִשְׁמַע אִשּׁוּר.
אֲנִי הָיִיתִי מוֹסִיפָה עוֹד קְצָת סֻכָּר, חָשַׁבְתִּי בְּלִבִּי וְלֹא
הֵעַזְתִּי לוֹמַר שֶׁלֹּא יַחְשְׁדוּ בִּי שֶׁאֵין מֵי הַכִּנֶּרֶת טְעִימִים לְחִכִּי.
תַּם הַשַּׁיִט וַאֲנַחְנוּ עַל שְׂפַת יָם כִּנֶּרֶת בְּמָקוֹם הַקָּרוּי "טַבְחָה".
עַד מְהֵרָה מָצָאנוּ עַצְמֵנוּ רְכוּבִים עַל גַּבֵּי הַפְּרָדוֹת עִם אֻכָּ־
פיהֶם הָרְחָבִים, וְ... עוֹלִים וּמְטַפְּסִים בִּשְׁבִילֵי עַקְלָתוֹן וּפָנֵינוּ
לְרֹאשׁ פִּנָּה. בַּדֶּרֶךְ נִפְתַּח כְּלוּב וְהַיּוֹנִים עֲזָבוּנוּ לְלֹא נְקִיפַת
מַצְפּוּן וַאֲפִילוּ לְלֹא אִחוּלֵי הַצְלָחָה בְּדַרְכֵּנוּ הַחֲדָשָׁה.

כַּאֲמוּר, הָיָה זֶה בַּחֹרֶף, וּבְאֵין כְּבִישִׁים קָשֶׁה הָיָה לַעֲבֹר
אֶת הַדֶּרֶךְ עַל נְחָלֶיהָ וּבִּצּוֹתֶיהָ, שֶׁבֵּין רֹאשׁ פִּנָּה לִמְטֻלָּה,
וּמַה גַּם לְ"חַייל כָּבֵד" כָּמוֹנוּ. נִשְׁאַרְנוּ, אֵיפֹא, לַחֲרֹוף בְּרֹאשׁ־
פִּנָּה וְאַבָּא נִתְמַנָּה מְמַלֵּא מְקוֹמוֹ שֶׁל מְנַהֵל בֵּית־הַסֵּפֶר שָׁם,
שִׂמְחָה וִילְקוֹמִיץ שֶׁשָּׁהָה אוֹתוֹ זְמַן בְּחוּ"ל.
     בָּאָבִיב, עִם הַסְּנוּנִיּוֹת הָרִאשׁוֹנוֹת, הִגַּעְנוּ לִמְטֻלָּה.

עוֹד בְּשִׁבְתֵּנוּ בְּרֹאשׁ פִּנָּה נָסַע אַבָּא, רָכוּב עַל חֲמוֹר ("הָרֶכֶב"
הַמְּקֻבָּל בְּיוֹתֵר בְּאוֹתָם יָמִים), לִמְטֻלָּה לִבְחֹן מִקָּרוֹב אֶת
מְקוֹם עֲבוֹדָתוֹ וּמְגוּרֵינוּ, הֶחָדָשׁ. בְּשׁוּבוֹ בִּשֵּׂר לָנוּ שֶׁקָּנָה
אֵצֶל אֶחָד הָאִכָּרִים עֶשֶׂר רְחֵלוֹת עִם טְלָיוֹתֵיהֶן. לשִׂמְחָה לֹא
הָיָה גְּבוּל וּמִיָּד יָשַׁבְנוּ כֻּלָּנוּ וְאַבָּא בָּרֹאשׁ, לַחְשֵׁב מַה יִּהְיֶה גֹּדֶל
הָעֵדֶר תּוֹךְ 3-2 הַשָּׁנִים הַקְּרוֹבוֹת. הַחֶשְׁבּוֹן הָיָה פָּשׁוּט וְקַל :
