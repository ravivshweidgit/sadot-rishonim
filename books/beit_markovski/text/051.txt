הָאֶבֶן , אֶת הַמַּשְׁפֵּךְ וְעוֹזֵר לַגַּרְעִינִים לָצֵאת מְעַט מְעַט מִתּוֹךְ
הַפֶּתַח שֶׁבְּתַחְתִּית הַמַּשְׁפֵּךְ - אֶל הָרֵחַיִם . עִם צֵאת אַחֲרוֹן
הַגַּרְעִינִים חִטָּה , תִּירָס , דּוּרָה אוֹ שְׂעוֹרָה - הָיָה הַטּוֹחֵן
דּוֹחֵף אֶת הַמּוֹט אֶל בֵּין שְׁלַבֵּי הַגַּלְגַּל הַמֵּנִיעַ - וְהַמְּכוֹנָה נֶע־
צֶרֶת . פָּשׁוּט וְקַל אֲבָל מַצְרִיךְ כַּמָּה שָׁעוֹת טוֹבוֹת לִטְחִינַת
שַׂק אֶחָד . גַּם אִיסּוּף הַקֶּמַח הָיָה פָּשׁוּט וְקַל . אַחֲרֵי שֶׁנִּגְרַף
וְהוּשַׂם בְּשַׂק , הָיָה הַטּוֹחֵן נִכְנָס בְּרַגְלָיו הַיְּחֵפוֹת ( בְּוַדַּאי רָחַץ
אוֹתָם אִי פַּעַם הֲלֹא מַיִם לֹא חָסְרוּ לוֹ ) אֶל תּוֹךְ תְּעָלַת הַקֶּמַח
וּמְטַאטֵא בְּפִיסַּת עוֹר בַּעַל־חַיִּים ( אוּלַי חָתוּל ) אֶת שְׁאֵרִית
הַקֶּמַח אֶל תּוֹךְ הַשַּׂק ( חֲבָל עַל כָּל טִפָּה ). נִשְׁמָע לֹא נָעִים
בְּיוֹתֵר אֲבָל ... הַלֶּחֶם - שֶׁנֶּאֱפָה לְאַחַר מִכֵּן בַּבַּיִת - הָיָה כְּבָר
מְשׁוּחְרָר מִכָּל מַגַּע יָדַיִם וְרַגְלַיִם וּשְׁאָר יְרָקוֹת .
הַטְּחִינָה , כָּאָמוּר , הָיְתָה אִיטִּית מְאֹד וְאִם עוֹד נִתְמַזֵּל
מַזָּלְךָ וְהָיָה תֹּר גָּדוֹל בַּטַּחֲנָה - שֶׁהֲרֵי גַּם שְׁכֵנֵינוּ מֵהַכְּפָרִים
הַסְּמוּכִים הָיוּ בָּאִים לִטְחֹן - הָיָה זֶה בִּלּוּי נָעִים בְּחֶבְרָה
טוֹבָה שֶׁל עַרָבִים וְעַרְבִיּוֹת . הֵם הָיוּ אוֹפִים פִּתּוֹת דַּקִּיקוֹת
טְעִימוֹת וַאֲנַחְנוּ סָעַדְנוּ עִמָּם יַחַד , מְשׂוֹחֲחִים וּמַאֲזִינִים לְסִפּוּרֵי
אֶלֶף לַיְלָה וְלַיְלָה שֶׁבְּפִיהֶם . לֹא אַחַת הָיִינוּ מְבַלִּים יוֹם שָׁלֵם
בִּטְחִינַת שַׂק קֶמַח וְגַם לִי הִזְדַּמֵּן לֵהָנוֹת מִבִּלּוּי כָּזֶה .
כָּל זֶה הָיָה בַּחֹרֶף , כְּשֶׁהַוָּאדִי הָיָה מָלֵא מַיִם וְהַזֶּרֶם בְּ־
מַפָּלִים חָזָק . אֲבָל בַּקַּיִץ הָיוּ הָעַרָבִים - שֶׁבִּרְשׁוּתָם הָיָה
מַעְיַן הַדַּרְדָּרָה מִמֶּנּוּ זָרְמוּ הַמַּיִם אֶל הַוָּאדִי וְאֶל הַטַּחֲנָה
שֶׁלְּיַד מְטֻלָּה - מַטִּים אֶת הַמַּיִם לְהַשְׁקוֹת שְׂדוֹתֵיהֶם . הַוָּאדִי
כִּמְעַט יָבֵשׁ , וְהַטַּחֲנָה שָׁבְתָה . הָיָה עָלֵינוּ לִנְדֹּד עִם שַׂקֵּי הַבָּר
הָעֲמוּסִים עַל גַּבֵּי אֲתֹנוֹת , אֶל הַחַצְבָּנִי לְטַחֲנַת הַכְּפָר
רִגִ'ר , מַהֲלַךְ שָׁעָתַיִם וְיוֹתֵר מִמְּטֻלָּה .
מִשְּׁהוּבָא הַקֶּמַח הַבַּיְתָה הָיָה זֶה כְּבָר מִתַּפְקִידָהּ שֶׁל
אִמָּא לֶאֱפוֹת לֶחֶם , וְלֹא הָיָה זֶה תַּפְקִיד קַל כְּלָל וְעִיקָר .
