בִּקַּשְׁתִּי לַעֲזֹר בִּמְקְצָת וְאִם כִּי הָיָה זֶה בִּתְחוּם זָר לִי לְגַמְרֵי -
נַעֲנֵיתִי לַבַּקָּשָׁה. וְאֵיךְ יָכֹלְתִּי לְסָרֵב כְּשֶׁנִּתְבַּקַּשְׁתִּי לְטַפֵּל בִּ־
קַיְּלָה גִּלְעָדִי שֶׁיָּלְדָה אֶת בְּנָהּ דּוּגַל וְהִיא חוֹלָה בְּדַלֶּקֶת רֵיאוֹת
(אַגַּב, הַיֶּלֶד מֵת בְּאוֹתוֹ שָׁבוּעַ שֶׁמֵּת אָבִיו יִשְׂרָאֵל גִּלְעָדִי). לֹא
הָיְתָה זוֹ מְשִׂימָה קַלָּה בְּגִילִי וּבְכִשׁוּרַי וּבוֹדַאי לֹא מִלֵּאתִי
תַּפְקִידִי כָּרָאוּי; אֶת הַגּוּגְל־מוּגְל (בַּבַּיִת קָרָאנוּ לָזֶה גּוֹג
וּמָגוֹג) שֶׁהִגַּשְׁתִּי לָהּ מִפַּעַם לְפַעַם כְּדֵי לְהָקֵל עַל הַשִּׁעוּל -
הִשְׁתַּדַּלְתִּי לִבְחוֹשׁ יָפֶה יָפֶה לְמַעַן לֹא יָצוּפוּ גּוּשֵׁי חֶלְמוֹן בַּכּוֹס
וְזֶה אוּלַי, הָיָה הַדָּבָר הַיְּחִידִי שֶׁעָשִׂיתִי טוֹב וְלִלְא הִסּוּס.
בְּפוּרִים הָרִאשׁוֹן לְשִׁבְתָּם בִּמְטֻלָּה, עָרְכוּ אַנְשֵׁי "הַשּׁוֹמֵר"
סְעוּדַּת פּוּרִים בְּאוּלָם בֵּית־הַפְּקִידוּת שֶׁנִּמְסַר לָהֶם לְמַטָּרָה זוֹ,
וַאֲנַחְנוּ הוּזְמַנּוּ לְהָסֵב עִמָּם לִסְעוּדָה, קְבוּצַת בַּחוּרִים וּבַחוּרוֹת
שֶׁעָשׂוּ בַּמּוֹשָׁבָה שָׁנָה רִאשׁוֹנָה לִפְנֵי עֲלוֹתָם לְהָקִים אֶת אִילַת־
הַשַּׁחַר - שֻׁתְּפוּ גַּם הֵם בִּסְעוּדַּת פּוּרִים זוֹ וְיַחַד מָנִינוּ
כְּ־50 אִישׁ, צִבּוּר גָּדוֹל בְּאוֹתָם זְמַנִּים, וְאֵין צָרִיךְ לוֹמַר שֶׁהַ־
שִׂמְחָה מְלֻוָּה, כָּרָגִיל, שִׁירָה וְרִקּוּדִים עַד אוֹר הַבֹּקֶר -
הָיְתָה רַבָּה.
בְּפֶסַח אָפִינוּ יַחַד מַצּוֹת. "תַּעֲשִׂיַּת" הַמַּצּוֹת הָיְתָה בְּבֵיתָם
שֶׁל אֶסְתֵּר וּצְבִי בֶּקֶר. בְּמִטְבָּחָם הַגָּדוֹל וּבוֹ תַּנּוּר גָּדוֹל וְטוֹב,
אָפִינוּ אֶת הַמַּצּוֹת. מוֹיְשֶׁה מוּגַיְּנִי מִקְּבוּצַת הָרוֹעִים, עָמַד
לְיַד הַתַּנּוּר וְאָפָה בִּמְוּמְחִיּוּת אֶת הַמַּצּוֹת שֶׁאֲנַחְנוּ, הַבָּנוֹת,
גִּלְגַּלְנוּ וְעָרַכְנוּ בְּמַעֲרוֹכִים שֶׁבְּיָדֵינוּ. עָשִׂינוּ עֲבוֹדָה יָפָה. הַמַּצּוֹת
הָיוּ עֲגֻלּוֹת, דַּקּוֹת וּטְעִימוֹת כְּמוֹהֶן לֹא הָיוּ וְאֵין עַד הַיּוֹם
הַזֶּה. גַּם עַל כַּשְׁרוּתָן הִקְפַּדְנוּ. אֶת הַבָּצֵק לַשְׁנוּ קִמְעָה קִמְעָה
לְמַעַן לֹא יַחְמִיץ וְנָקַטְנוּ גַּם בִּשְׁאָר כָּלְלֵי כַּשְׁרוּת.

וּבִשְׁמִירָה. בַּשָּׂדוֹת - שָׁמְרוּ בְּקַיִץ בִּרְכִיבָה עַל סוּסוֹתֵיהֶם.
