עַל אִמָּא לֶאֱפוֹת לֶחֶם וְלָרֹב , גַּם לְבַשֵּׁל . לִפְתִיחַת הַתַּבּוּן הָיָה
עָלֶיהָ לִגְרֹף אֶת שִׁכְבַת הָרֶמֶץ מֵעַל הַמִּכְסֶה , לְהָסִירוֹ וּלְהַכְנִיס
לְעֻמְקוֹ שֶׁל הַתַּבּוּן אֶת כִּכְרוֹת הַבָּצֵק לַאֲפִיָּה אוֹ קִדְּרַת
עֲדָשִׁים וּבּוּרְגּוּל לְבִשּׁוּל . אֵין צֹרֶךְ לוֹמַר שֶׁהָיְתָה זוֹ עֲבוֹדָה
קָשָׁה וּמְפָרֶכֶת .
כָּךְ קִיַּמְנוּ כֻּלָּנוּ , לְמִגָּדוֹל וְעַד קָטֹן , אֶת הַכָּתוּב : "בְּזֵעַת
אַפֶּיךָ תֹּאכַל לֶחֶם ".

הַבּוּרְגּוּל — גַּם הוּא לֹא עָלָה עַל שֻׁלְחָנֵנוּ כְּשֶׁהוּא מוּבָא
מִן הַצַּרְכָּנִיָּה . הָיְתָה זוֹ "תַּעֲשִׂיָּה" כְּשֶׁלְּעַצְמָהּ . הָיִינוּ מְבַשְּׁלִים
אֶת הַחִטָּה בְּדוּד הַנְּחֹשֶׁת — לָמָּה לֹא שִׁמֵּשׁ אוֹתָנוּ כְּלִי
נִפְלָא זֶה — שָׁעוֹת אֲרֻכּוֹת עַד הֱיוֹתָהּ רַכָּה וּתְפוּחָה , מְיַבְּשִׁים
עַל מַחְצָלוֹת וְגוֹרְסִים אַחַר־כָּךְ בְּגֻ'רוּשָׁה (רֵיחַיִם שֶׁל יָד עֲשׂוּיִים
אֶבֶן בָּזֶלֶת . כַּיּוֹם — מְשַׁמְּשִׁים לְקִשּׁוּט גִּנּוֹת ). מְנַפִּים וּמַפְרִי
דִים בֵּין גְּרִיסֵי הַבּוּרְגּוּל הַדַּקִּים וְהַגַּסִּים יוֹתֵר , לְשִׁמּוּשׁ שׁוֹ־
נֶה . אָהַבְנוּ לִזְלֹל מֵהַחִטָּה הַמְּבֻשֶּׁלֶת בְּטֶרֶם יֻבְּשָׁה .

קִים הָאֲהוּבִים עָלַי בְּיוֹתֵר , הַחֵל בְּאִכּוּף הָאָתוֹן בְּאֹוכף רָחָב
וְכָבֵד , מִשְּׁנֵי עֲבָרָיו תּוֹלִים אַרְגָּזִים וּבָהֶם אַרְבָּעָה פַּחִים , וְאַתְּ
רוֹכֶבֶת לְךָ בְּנִיחוּתָא מֵעַל כָּל הַכְּבוּדָה הַזּוּ בְּמוֹרַד הַגִּבְעָה

כְּמוֹ , לְהַבְדִּיל , בִּכְבִישֵׁי הָאָרֶץ הַיּוֹם , אֲבָל מִי מוּכָן לָתֵת לוֹ ?
