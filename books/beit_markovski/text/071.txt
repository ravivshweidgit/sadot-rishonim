הַשֶּׁטַח הַקָּרוּי "שַׁרְקִיָּה", חֻלַּק לַחֲלָקוֹת שָׁווֹת וְכָל מֶשֶׁק
קִבֵּל אֶת חֶלְקָתוֹ לְפִי הַגְרָלָה וְהָיָה קוֹצֵר (בְּחֶרְמֵשִׁים) אֶת
הַתִּלְתָּן וּמְיַבְּשׁוֹ לְחָצִיר.
נָסַעְנוּ בְּעֲגָלָה לְאוֹתָהּ שַׁרְקִיָּה, יְהוּדָה נוֹהֵג בַּסּוּסִים וַאֲנַחְנוּ
עִם עוֹד יְלָדִים מֵחֲבֵרֵינוּ, מְמַלְּאִים אֶת הָעֲגָלָה. לְפֶתַע הִתְ־
חשֶׁק לַסּוּסִים לִדְהֹר בְּמוֹרַד הַמּוֹשָׁבָה בְּכִוּוּן הַמרג', מִבְּלִי
שֶׁיְּהוּדָה הִצְלִיחַ לַעֲצֹר בַּעֲדָם. הָעֲגָלָה הֵחֵלָּה לְהִתְפָּרֵק ; הַלּוּחַ
הַקִּדְמִי נָפַל עַל רַגְלֵי הַסּוּסִים וְהִדהִיר אוֹתָם עוֹד יוֹתֵר ; גַּם
יֶתֶר הַלּוּחוֹת אִבְּדוּ אֶת מְקוֹמָם הַקָּבוּעַ וְהֵחֵלּוּ מְרַקְּדִים
וּמתכנְּנִים לִקְפֹּץ וּלְהִשְׁתַּחְרֵר מֵהָעֲגָלָה – כְּשֶׁמְּיֻדָּעֵנוּ קֶסֶלְמַן
צָץ וְהוֹפִיעַ מֵאֵי־שָׁם אֶל מוּל פְּנֵי הַסּוּסִים וְעָצַר בַּעֲדָם. שֶׁטֶף
מִלּוֹת הַנְּזִיפָה וְהַתּוֹכֵחָה בְּנוּסַח: "אֵיךְ הֵעַזְתָּ לְמַלֵּא אֶת
הָעֲגָלָה יְלָדִים כְּשֶׁאֵינְךָ יוֹדֵעַ לִנְהֹג בַּסּוּסִים", וְאֵיךְ... וְאֵיךְ -
זָרַם, כְּמַעְיָן הַמִּתְגַּבֵּר מִפִּיו שֶׁל קֶסֶלְמַן. שָׁכַח מְיֻדָּעֵנוּ הַנִּכְבָּד
שֶׁגַּם לוֹ קָרְתָה, לֹא מִכְּבָר, תַּקָּלָה (וּבוודַאי לֹא אַחַת). וּבִכְלָל,
מֶה הָיָה עַל הַנִּסִּים לוּ יָשַׁבְנוּ כֻּלָּנוּ בַּבַּיִת וְלֹא "הֵעַזְנוּ".

תִּרְצָה

לֹא תָּמִיד הָיָה רוֹפֵא בַּמּוֹשָׁבָה. וּכְשֶׁלֹּא הָיָה - הָיִינוּ נִזְקָּ־
קִים לְרוֹפְאִים שְׁכֵנִים. שְׁנֵי רוֹפְאִים, עַרְבִים נוֹצְרִים, הָיוּ בְּ־
שְׁכֵנוּתֵנוּ. הָאֶחָד, ד"ר אַמְגִ'ילָה, בּוֹגֵר אוּנִיבֶרְסִיטָה צָרְפָתִית
בְּבֵירוּת, לוֹבֵשׁ פְּרֶנְגִ'י וְחוֹבֵשׁ תַּרְבּוּשׁ מְגֹהָץ, הָיָה בָּא לַמּוֹשָׁבָה
פַּעֲמַיִם בַּשָּׁבוּעַ מִמְּקוֹם מְגוּרָיו גִ'דֵידָה "הָעִיר", כְּשֶׁהוּא רָכוּב
עַל סוּסוֹ, הַשִּׁמְשִׁיָּה בְּיָדוֹ וְכֻלּוֹ אוֹמֵר הַכָּרַת עֵרֶךְ עַצְמוֹ.
הַשֵּׁנִי - ד"ר אַמִין, בּוֹגֵר הַקּוֹלֶג' הָאָמֵרִיקָאִי בְּבֵירוּת וְלִ־
זְכוּתוֹ גַּם תּוֹסֶפֶת הִשְׁתַּלְּמוּת בְּאָמֵרִיקָה. אָדָם אִינְטֶלִיגֶנְטִי,
שָׁקֵט וְנֶחְמָד כְּכָל אַנְשֵׁי כַּפְרוֹ דִיר־מִימַס, הָיָה הַדּ"ר אַמִין.
