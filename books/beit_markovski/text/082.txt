מַיִם וְהַקָּרִים . הָיָה עַל הַשּׁוֹמְרִים לַעֲבֹר מֵאֲחוֹרֵי הַקִּיר וְלִבְדֹּק
אֶת הַדְּלָתוֹת הַקְּרוּעוֹת בְּכָל חָצֵר לְכִוּוּן הַחַכּוּרָה זְרוּעַת
הַסְּלָעִים — אִם לֹא נִפְרְצָה אֵיזוֹ מֵהֶן . בְּלֵילוֹת גֶּשֶׁם הָיוּ
נַעֲלֵיהֶם וְהָעַבָּאיָה שֶׁעֲלֵיהֶם , סְפוּגִים מַיִם . הָיוּ נִכְנָסִים , בֵּין
סִבוּב לְסִבוּב אֶל בִּקְתַּת הַתֶּבֶן שֶׁלָּנוּ לְהִתְחַמֵּם וּלְיַבֵּשׁ
קְצָת אֶת לְבוּשָׁם . אִמָּא הִשְׁגִּיחָה בְּכָךְ וְהָיְתָה מַקְפִּידָה שֶׁתָּמִיד
יִמָּצְאוּ שָׁם שְׁנֵי פַּחִים לִישִׁיבָה "נוֹחָה" בַּעֲבוּרָם .

   פַּעַם , הִתְנַסֵּיתִי גַּם אֲנִי בְּמִין סִבוּב כָּזֶה מֵאֲחוֹרֵי הַקִּיר .
אָמְנָם בְּלֵיל קַיִץ חַם וְלֹא בְּלַיְלָה גָּשׁוּם וְאַף־עַל־פִּי־כֵן הָיְתָה
זוֹ הַעְזָה מִצִּדִּי .
   וְאֵיךְ קָרָה הַדָּבָר ? פָּשׁוּט מְאֹד , עֶרֶב אֶחָד יָצָאתִי עִם
חֲבֶרְתִּי לְטַיֵּל בִּרְחוֹבָהּ שֶׁל הַמּוֹשָׁבָה . פָּגַשְׁנוּ בַּשּׁוֹמֵר שַׁבְּסיל
(שַׁבְּתַאי — שַׁבְּתִיל וּבְהֶבְרָה אַשְׁכְּנַזִּית — שַׁבְּסִיל) וְהוּא
נִלְוָה אֵלֵינוּ . בִּקְצֵה הָרְחוֹב אָמַר שֶׁעָלָיו לַעֲשׂוֹת דּוּר (סִבוּב) .
תּוֹךְ כְּדֵי שִׂיחָה , נִזְרְקָה לֶחָלָל הַצָּעָה שֶׁאֲנַחְנוּ הַבָּנוֹת , נַעֲשֶׂה
אֶת הַדּוּר . אֲנִי הִסְכַּמְתִּי מִיָּד בִּתְנַאי שֶׁאֵלֵךְ מֵאֲחוֹרֵי הַקִּיר
הַמֻּכָּר לִי יוֹתֵר , בַּצַּד בּוֹ גַּרְנוּ אֲנַחְנוּ , וַחֲבֶרְתִּי שֶׁגָּרָה מִמּוּל —
גַּם לָהּ יִהְיֶה הַקִּיר הַמֻּכָּר . שַׁבְּסִיל נָתַן לִי מַשְׁרוֹקִית (עַל
כָּל צָרָה שֶׁלֹּא תָּבוֹא) וַאֲנִי פָּתַחְתִּי מִיָּד בַּמְּשִׂימָה . הָלַכְתִּי
לְאִטִּי וְלֹא פָּסַחְתִּי עַל אַף דֶּלֶת אַחַת מִבְּלִי לִבְדֹּק וְלִרְאוֹת
אִם אָמְנָם הִיא נְעוּלָה . אֲנִי מוֹדָה וּמִתְוַדָּה שֶׁ"הַטִּיּוּל" בְּ־
חֶשְׁכַּת הַלַּיְלָה בֵּין הַסְּלָעִים כְּשֶׁאוֹ־טוֹ־טוֹ יְזַנֵּק מִישֶׁהוּ מֵאֲחוֹרֵי
הַסֶּלַע — הָיָה "קְצָת" מַפְחִיד וְאַף־עַל־פִּי־כֵן לֹא עָשִׂיתִי
מְלַאכְתִּי רְמִיָּה וְלֹא פָסַחְתִּי כַּאֲמוּר , עַל בְּדִיקַת כָּל הַדְּלָתוֹת .
הִשְׁתַּדַּלְתִּי לִהְיוֹת שְׁקֵטָה וְלֹא לְאַבֵּד מֵאֹמֶץ הַלֵּב שֶׁסִּגַּלְתִּי
לִי מֵאָז עָמַדְתִּי עַל דַּעְתִּי . הַסִּבּוּב נִגְמַר בְּטֹב . אִישׁ לֹא
זִנֵּק מֵאֲחוֹרֵי הַסְּלָעִים . כְּשֶׁהִגַּעְתִּי לִמְקוֹם הַמִּפְגָּשׁ בִּקְצֵהוּ
