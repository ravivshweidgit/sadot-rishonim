הַיְינוּ מִתְיַעֲצִים וְקוֹבְעִים בְּיַחַד מַה לִּזְרֹעַ בַּשָּׂדֶה זֶה וּמַה
בְּאַחֵר.
גַּם אִם הֶאֱרַכְתִּי , כַּאֲמוּר , עָלַי עוֹד לְסַפֵּר עַל תְּלִישׁ הַחֻמְצָה ,
שֶׁזֶּהוּ פֶּרֶק מְיֻחָד בְּמִינוֹ וְאֵין לִפְסֹחַ עָלָיו.
לֹא זוֹ בִּלְבַד שֶׁגִּבְעוֹלֵי הַחֻמְצָה הַקָּשִׁים וְהַחַמְצַמְצִים סוֹדְ־
קִים בְּכַפּוֹת הַיָּדַיִם עַד זוֹב דָּם וְצוֹרְבִים בַּחֲמִיצוּתָם — יָדֵינוּ
אָנוּ , כְּבָר הָיוּ מְחֻסָּנוֹת נֶגֶד צָרְבָנִים אֵלּוּ — יֵשׁ לִתְלֹשׁ בְּעוֹד
הַטַּל בַּשָּׂדֶה וְהַצְּמָחִים לַחִים , שֶׁאִם לֹא כֵן נוֹשְׁרִים רַבִּים מֵהַ־
תַּרְמִילִים וְגַרְעִינֵיהֶם וְנִשְׁאָרִים בַּשָּׂדֶה. הַיְינוּ תּוֹלְשִׁים בְּשָׁעוֹת
הַבֹּקֶר הַמֻּקְדָּמוֹת וְלְעִתִּים , אִם מִיקוּם הַשָּׂדֶה אִפְשֵׁר זֹאת ,
גַּם בְּשָׁעוֹת הָעֶרֶב וְלָנִים בַּשָּׂדֶה.

תְּלִישׁ הַחֻמְצָה הָאַחֲרוֹן שֶׁלִּי בִּמְטֻלָּה (בַּשָּׁנָה שֶׁלְּאַחֲרֶיהָ
כְּבָר הָיִיתִי בְּאֵילַת־הַשַּׁחַר ), הָיָה בְּזִיקְרָה. שָׂדֶה בְּקִרְבַת הַכְּפָר
אִיבֶּל , עִם רֶדֶת הָעֶרֶב יָצְאוּ יְהוּדָה , עִמָּנוּאֵל וַאֲנִי. שְׁכֵנֵנוּ
הָאִכָּר , נִלְוָּה גַּם הוּא אֵלֵינוּ לִתְלֹשׁ בְּשָׂדֵהוּ. יָרַדְנוּ בְּרִיצָה ,
בְּצִחְצוּקִים וְקוֹלָנוּת אֶל הַוָּאדִי. עוֹד אָנוּ מְדַלְּגִים מֵאֶבֶן לְאֶבֶן
וְחוֹצִים אֶת מֵימֵי הַנַּחַל בְּדַרְכֵּנוּ אֶל הַשָּׂדֶה שֶׁמֵּעֵבֶרוֹ — נִתְקַף
שְׁכֵנֵנוּ (שַׁמְנְמַן וְכַרְסְתָּנִי בְּמִקְצָת ) פַּחַד , וְהֵחֵל מְבַקֵּשׁ שֶׁלֹּא
נָקִים רַעַשׁ בְּאָמְרוֹ : תּוּמֶער אַגְנֶב פוּפְּט דָּארְטֶען (שֶׁמָּא גַּנָּב
יָשֵׁן שָׁם ) מוּבָן שֶׁהוּא הִשִּׂיג אֶת הֵהִיפֵךְ מִמַּה שֶּׁבִּקֵּשׁ , כִּי
מִיהוּ שֶׁיָּכוֹל לְהִתְאַפֵּק וְלֹא לְהִתְפַּקֵּעַ מִצְּחוֹק לְשֵׁמַע מִשְׁפָּט
מַבְרִיק כָּזֶה ? אוֹתוֹ בַּרְנָשׁ כְּבָר לֹא יָצָא עִמָּנוּ לִתְלֹשׁ חֻמְצָה
בַּלַּיְלָה.

הִגַּעְנוּ לַשָּׂדֶה , תָּלַשְׁנוּ כַּמָּה שָׁעוֹת טוֹבוֹת עַד שְׁקִיעַת הַיָּרֵחַ
בְּ־ 11 בְּעֵרֶךְ , תָּקַעְנוּ שְׁנֵי מוֹטוֹת בָּאֲדָמָה , מָתַחְנוּ עֲלֵיהֶם שַׂק
אֶחָד שֶׁנִּמְצָא לָנוּ בַּשָּׂדֶה — וְהִכְרַזְנוּ שֶׁזֶּהוּ אֹהֶל לִלְינַת לַיְלָה.
חֵלֶק מֵהַשָּׁעוֹת הַמְּעֻטּוֹת שֶׁנּוֹתְרוּ לָנוּ לְשֵׁינָה "בִּזְבַּזְנוּ " בְּצִיחְ־
