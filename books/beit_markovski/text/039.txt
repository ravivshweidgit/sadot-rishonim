שֶׁהָיָה יְדִיד בֵּיתֵנוּ , בָּא מִיָּד וְהֵבִיא עִמּוֹ כֶּסֶף , לִמְנֹעַ כָּל עִכּוּב ,
וְצִוָּה לִנְסֹעַ מִיָּד קָהִירָה לְקַבָּלַת זְרִיקוֹת נֶגֶד כַּלֶּבֶת . תָּאֲרוּ
לְעַצְמְכֶם מָה הָיוּ "מוֹסְדוֹת" הָרְפוּאָה ( אִם בִּכְלָל הָיוּ כָּאֵלֶּה )
בָּאָרֶץ , אִם לְשֵׁם דָּבָר פָּעוּט כָּזֶה צָרִיךְ הָיָה לְהַפְלִיג עַד קָהִיר
( אֲנִי חוֹשֶׁבֶת שֶׁכַּיּוֹם הֵם יְכוֹלִים לָבוֹא אֵלֵינוּ לְטִפּוּלִים רְפוּ־
אִיִּים . אַדְּרַבָּא , שֶׁיָּבֹאוּ ).
וּבְכֵן , נוֹסְעִים לְקָהִיר . וְאֵיךְ נוֹסְעִים ? בְּוַדַּאי , לֹא בְּמָטוֹס .
בִּרְכִיבָה עַל פְּרִידוֹת עַד צִידוֹן , שָׁם לָנִים וּלְמָחֳרָת מַמְשִׁיכִים
בְּדִלְגֶ'נְס ( מִין כִּרְכָּרָה ) עַד בֵּירוּת , בְּאֳנִיָּה לְפוֹרט־סְעִיד וּבְ־
רַכֶּבֶת לְקָהִיר . טִיּוּל לֹא רַע וּבְכָל כְּלֵי הַתַּחְבּוּרָה .
לְמַעַן הָאֱמֶת יֵשׁ לוֹמַר שֶׁלֹּא הָיָה הֶכְרֵחַ לִנְסֹעַ דַּוְקָא לְקָהִיר
וְאֶפְשָׁר הָיָה לְקַבֵּל אוֹתוֹ טִפּוּל בְּאִיסְטַנְבּוּל עַל חֶשְׁבּוֹן מֶמְשַׁל־
תֵּנוּ דָּאָז , מֶמְשֶׁלֶת תּוּרְכִּיָּה ; אֲבָל מִשּׁוּם מָה נִקְבַּע שֶׁאֵין
לִסְמֹךְ עַל הַתּוּרְכִים בְּעִסְקֵי בְּרִיאוּת וּבְמִצְרַיִם הַנִּסְיוֹב נֶגֶד
כַּלֶּבֶת בְּוַדַּאי טוֹב וְיָעִיל יוֹתֵר . אַחֲרֵי הַכֹּל מִצְרַיִם הָיְתָה זָרָה
וּרְחוֹקָה לָנוּ וְאִילוּ תּוּרְכִּיָּה מֶמְשֶׁלְתֵּנוּ שֶׁלָּנוּ וְלֵית מַאן

הַנְּסִיעָה עַל חֶשְׁבּוֹן הַמֶּמְשָׁלָה וְלִנְסֹעַ ע"ח הַבַּיִת , לַמְרוֹת שֶׁאף
פַּעַם הוּא לֹא סָבַל מֵעֹדֶף כֶּסֶף .

רְפוּאִי , אֲבָל בְּדִיעֲבַד נֶהֱנוּ הַשְּׁנַיִם מִטִּיּוּל יָפֶה וּמְעַנְיֵן וְלֹא
הֶחְמִיצוּ כָּל הִזְדַּמְּנוּת לְבַקֵּר וְלִרְאוֹת כְּכָל שֶׁנִּיתָּן וְזֶה הָיָה לֹא
מְעַט . עוֹד יָמִים רַבִּים לְאַחַר שׁוּבָם הָיוּ מְשִׂיחִים וּמְסַפְּרִים
בִּנְסִיעַת מִצְרַיִם וּבְכָל הַנִּפְלָאוֹת שֶׁרָאוּ שָׁם לְמִן הַפִּירָמִידוֹת
