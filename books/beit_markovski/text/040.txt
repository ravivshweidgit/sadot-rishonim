אֶת יְמֵי הַהֻלֶּדֶת נָהַגְנוּ לְצַיֵּן תָּמִיד בְּחוּג הַמִּשְׁפָּחָה בִּלְבַד
וּבְאֹרַח צָנוּעַ בְּיוֹתֵר. עִקַּר הָאֵרוּעַ הָיוּ הַמַּתָּנוֹת (לָרֹב -
סְפָרִים) שֶׁהָיוּ הַהוֹרִים מַנִּיחִים עַל כִּסֵּא לְיַד מִטָּתוֹ שֶׁל חֲתַן
יוֹם הַהֻלֶּדֶת אֲשֶׁר הָיָה מַשְׁכִּים קוּם וְלִפְעָמִים עַד כִּי הָיָה
מְגַשֵּׁשׁ בַּאֲפֵלֻלִּית הַבֹּקֶר וּמְנַחֵשׁ מַה מֻּנָּח עַל הַכִּסֵּא. הַ־
פַּעַם - חָרַגְנוּ מֵהַמְּקֻבָּל וְהֻזְמְנוּ כָּל יַלְדֵי הַכִּתָּה עַל מוֹרָם.
"מַסְמְרֵי" הַחֲגִיגָה הָיָה מִין לַהֲטוּט שֶׁהִצִּיג הַמּוֹרֶה בְּעֶזְרַת
"מַכְשִׁיר" שֶׁהוּבָא יָשָׁר מִמִּצְרַיִם. וְהַמַּכְשִׁיר לֹא הָיָה אֶלָּא פָּנָס
חַשְׁמַלִּי קָטָן בַּעַל וו שֶׁאֶפְשָׁר הָיָה לִתְלוֹתוֹ עַל כִּיס הַבֶּגֶד,
הַסּוֹלְלָה - מְחֻבֶּרֶת בְּחוּט אֶל הַפָּנָס. הַמּוֹרֶה כִּבָּה אֶת
הָאוֹר בַּחֶדֶר וְהוֹדִיעַ לַיְלָדִים: "אֶסְפֹּר עַד שָׁלֹשׁ וְאָז יוֹפִיעַ
מִין יָרֵחַ קְטַנְטַן, עָגֹל וּמֵאִיר". הוֹשִׁיב אֶת יְהוּדָה מוּל הַ־
יְלָדִים, תָּלָה אֶת הַפָּנָס עַל חָזֵהוּ, הַסּוֹלְלָה בְּכִיסוֹ וְלָחַשׁ לוֹ
לִלְחוֹץ עַל הַכַּפְתּוֹר עִם הַקְּרִיאָה שָׁלֹשׁ. וְהִנֵּה הִתְרַחֵשׁ הַ"פֶּלֶא"
כַּמֻּבְטָח, לְהַפְתָּעָתָם וְלְשִׂמְחָתָם שֶׁל הַיְלָדִים אֲשֶׁר בִּקְּשׁוּ
לַחְזֹר עַל מַעֲשֵׂה הַנִּסִּים וְכַמּוּבָן נֶעֱנוּ.

וְאִם כְּבָר סִפַּרְתִּי עַל יוֹם הֻלֶּדֶת אֶחָד - אֲסַפֵּר גַּם
עַל שניִ.
בֹּקֶר אֶחָד הָיִיתִי אֲנִי זוֹ שֶׁצִּפְּתָה וְהִתְאַכְזְבָה כְּשֶׁהַכִּסֵּא
שֶׁלְּיַד מִטָּתָהּ הָיָה רֵיק. אַךְ לְהַפְתָּעָתִי הִגִּיעָה לְאָזְנַי פְּעִיָּתָהּ
שֶׁל עֵז לְיַד דֶּלֶת הַבַּיִת. הָעֵז נִגְרְרָה פְּנִימָה וְזוֹ הָיְתָה הַמַּתָּנָה
שֶׁקִּבַּלְתִּי. הַשִּׂמְחָה הָיְתָה גְּדוֹלָה ; בַּחֲלוֹם הַוָּרֹד בְּיוֹתֵר לֹא
יָכֹלְתִּי לִרְאוֹת מַתָּנָה יָפָה מִזּוֹ. יוֹם יוֹם, בְּצָהֳרֵי הַיּוֹם, הָיִיתִי
הוֹלֶכֶת אֶל הַגֹּרֶן שָׁם נֶאֶסְפוּ הַצֹּאן לַחֲלִיבָה ; הָרוֹעֶה (עַרְבִי)
הָיָה חוֹלֵב לְתוֹךְ הַכְּלִי שֶׁהֵבֵאתִי עִמִּי מִקְצָת מֵחֲלֵב הָעֵז
כְּכוֹס בְּעֵרֶךְ, צָרִיךְ הֲלֹא לְהַשְׁאִיר לַגְּדִי, וְהָרוֹעֶה, גַּם הוּא אֵינוֹ
