הָרִאשׁוֹנוֹת נִטַּשְׁטַשׁ הֶבְדֵּל זֶה כִּמְעַט לַחֲלוּטִין. תָּמִיד הָיוּ
יַחַד. גַּם הַלְּבוּשׁ לֹא הָיָה שׁוֹנֶה וְכַאֲמוּר, נֶעְלַם גַּם הֶבְדֵּל הַגִּיל.

לֵיצָנֵי הַדּוֹר קָרְאוּ לָהֶם "יַשְׂבָּסִיל מְלַמֵּד", וְעַל שׁוּם מָה ?
שְׁנֵי בְּרָנָשִׁים הָיוּ בַּמְּטֻלָּה, שַׁבְּתִיל (וּבְהֶבְרָה אַשְׁכְּנַזִּית
שַׁבְּסִיל) וּמְלַמֵּד. הָיוּ אֵלֶּה אָב וּבְנוֹ, אִכָּרִים, תִּמְהוֹנִים בִּמְק־
צָת. שׁוֹנִים וְנִבְדָּלִים הָיוּ מִשְּׁאַר הָאִכָּרִים, וְאִם כִּי עֲבוֹדַת
הָאֲדָמָה הָיְתָה זָרָה לָהֶם וּבְקֹשִׁי לָמְדוּ כֵּיצַד עוֹשִׂים בִּ־
מְלָאכָה - עִבְּדוּ שְׂדוֹתֵיהֶם בְּמוֹ יְדֵיהֶם וּבְלִי חָרָתִים עַרָבִים
וְעַל כָּךְ הָיְתָה גַּאֲוָתָם. וְכַאֲשֶׁר הָיָה בָּא פָּקִיד הַבָּרוֹן לַמּוֹשָׁבָה -
הָיוּ הַשְּׁנַיִם מִתְיַצְּבִים לְפָנָיו וּמַצְהִירִים, כַּאֲשֶׁר הָאֶחָד מְאַשֵּׁר
וּמְחַזֵּק דִּבְרֵי קוֹדְמוֹ (כְּדוּגְמַת הַשַּׁדְּכָן וְעוֹזְרוֹ), בַּזּוּ הַלָּשׁוֹן :
"אַלְעַיין גִּיחְלִישֶׁעט, אַלְעַיין גִּיחְסִידֶעט, אַלְעַיין גִּירְגֶ'דֶעט".
(בְּעַצְמֵנוּ תָּלַשְׁנוּ, בְּעַצְמֵנוּ קָצַרְנוּ, בְּעַצְמֵנוּ הוֹבַלְנוּ הַגֹּרְנָה).
וּמִתּוֹךְ שֶׁהָיוּ תָּמִיד יַחַד וְאֵין הָאֶחָד זָז בְּלִי הַשֵּׁנִי – "הִלְבִּישׁוּ"
אֶת שְׁמוֹתֵיהֶם לְתִרְצָה וְאֶזְרָח שֶׁגַּם הֵם לֹא זָזוּ הָאֶחָד בְּלִי
הַשֵּׁנִי.
גַּם לְבֵית־הַסֵּפֶר הָלְכוּ יַחַד וְלְאוֹתָהּ כִּתָּה. אַגַּב, אַבָּא לֹא הָיָה
מֵחֲסִידֵי גַּנֵּי הַיְלָדִים, הַגַּנָּנוֹת שֶׁל אָז (אִם בִּכְלָל הָיוּ) לֹא הָיוּ
דֵּי טוֹבוֹת בְּעֵינָיו בִּמְיֻחָד הָעִבְרִית שֶׁבְּפִיהֶן, וְלָכֵן לֹא נִשְׁלַחְנוּ
לְגַן. וְאוּלָם כְּדֵי שֶׁלֹּא נִהְיֶה בְּטֵלִים, חָלִילָה, מִלִּמּוּדִים כָּל־
שֶׁהֵם – הָיִינוּ מְלַמְּדִים הַגָּדוֹל אֶת הַצָּעִיר מִמֶּנּוּ, קְרוֹא וּכְתֹב.
וְכָךְ נִכְנַסְנוּ לְבֵית הַסֵּפֶר "גְּאוֹנִים" מֻשְׁלָמִים.

תִּרְצָה הָיְתָה שַׁקְדָנִית וַחֲרוּצָה בַּלִּמּוּדִים וְגַם בַּעֲבוֹדָה.
וְאָזִיקָה – אָהַב לְהִתְפַּנֵּק בִּמְקְצָת וְהָיָה מְנַצֵּל אֶת שַׁקְדָנוּתָהּ
וְנֶעֱזָר לְעִתִּים בְּשִׁעוּרִים שֶׁכְּבָר הֵכִינָה. לְיָמִים, כְּשֶׁעָבְרָה הַ־
מִשְׁפָּחָה (רַק הַהוֹרִים עִם שְׁנֵי הַקְּטַנִּים) לִירוּשָׁלַיִם
