דּוֹת , עַל הָרִצְפָּה סָבִיב מַכְתֵּשׁ לִכְתּוֹשׁ סֻכָּר . אֲפִילוּ הַסֻּכָּר
הָיָה אָז שׁוֹנֶה וְהָיָה מַגִּיעַ בְּגוּשִׁים גְּדוֹלִים בְּצוּרַת קוֹנוּס . גּוּשׁ
כָּזֶה קָרוּי הָיָה "רֹאשׁ סֻכָּר" וְהָיוּ מְקַצְּצִים אוֹתוֹ בְּקוֹפִיץ
לִגְבִישִׁים קְטַנִּים לְשִׁמּוּשׁ רָגִיל , וְלַאֲפִיָּה - כּוֹתְשִׁים וּמְנַפִּים
בְּנָפָה . אַגַּב , הַמַּכְתֵּשׁ הֶעָשׂוּי נְחֹשֶׁת הָיָה כְּלִי שִׁמּוּשִׁי בְּיוֹתֵר בְּכָל
בַּיִת , הָיוּ כּוֹתְשִׁים בּוֹ סֻכָּר , מֶלַח , קִנָּמוֹן , פִּלְפֵּל , שֻׁמְשׁוּם
לַאֲפִיָּה וּמָה לֹא . אִמָּא הָיְתָה מְיַבֶּשֶׁת וְכוֹתֶשֶׁת קְלִיפּוֹת תַּפּוּ־
זִים לְאָבָקָה דְּקִיקָּה שֶׁשִּׁמְּשָׁה לָהּ תַּבְלִין רֵיחָנִי לְעֻגּוֹת .
גָּמַרְנוּ לִכְתּוֹשׁ וְיָשַׁבְנוּ - גַּם זֹאת עַל הָרִצְפָּה - סָבִיב
גִּיגִית נְחֹשֶׁת גְּדוֹלָה וּבָהּ הַרְבֵּה חֶלְמוֹנֵי בֵּיצִים וְסֻכָּר וְנצ-
טווינוּ לִבְחוֹשׁ בְּכַפּוֹת שֶׁבְּיָדֵינוּ עַד הֱיוֹת הָעִסָּה לְבָנָה וּסְמִי־
כָה . הָאִמָּהוֹת , כָּל אַחַת צַלַּחַת וּמַזְלֵג בְּיָדָהּ , דּוֹפְקוֹת וּמַקְצִי־
פוֹת אֶת חֶלְבּוֹנֵי הַבֵּיצִים וְעוֹשׂוֹת בַּמְּלָאכָה בִּזְרִיזוּת צַלַּחַת
אַחֲרֵי צַלַּחַת עַד תֹּם כָּל הַחֶלְבּוֹנִים וְהֵם הָיוּ רַבִּים . הַכֹּל לְתוֹךְ ,
הַגִּיגִית , בָּחוּשׁ לְמַשְׂעִי (מִי צָרִיךְ מִיקְסֵר), וְיוֹצְקִים אֶת הָעִסָּה

אוֹ בְּחֹמֶר הַסָּקָה דּוֹמֶה , קוֹנְדִיטוֹרְיָה מְשֻׁכְלֶלֶת כָּזוּ לֹא תִמְצָא
בְּיָמֵינוּ אֵלֶּה .
בְּנוֹסָף לְעֻגּוֹת הוּכְנָה גַּם סְעֻדַּת שַׁבָּת מְיֻחֶדֶת וּמִשְׁפָּחוֹת
אֲחָדוֹת מֵהַמְּקֹרָבוֹת בְּיוֹתֵר סָעֲדוּ עַל שֻׁלְחָנֵנוּ בְּלֵיל הַשַּׁבָּת
וּבְצָהֳרֵי יוֹם הַשַּׁבָּת .

וְאוּלָם הַדָּבָר הַגָּדוֹל בְּיוֹתֵר הָיָה הֲלִיכָתָהּ שֶׁל אִמָּא לְבֵית
הַכְּנֶסֶת לִרְאוֹת וְלִשְׁמֹעַ אֶת בְּנָהּ בְּכוֹרָהּ עוֹלֶה לַתּוֹרָה .
אִמָּא הָיְתָה נִרְגֶּשֶׁת מְאֹד בְּלֶכְתָּהּ . אִם זִכְרוֹנִי אֵינוֹ מַטְעֵנִי
הָיְתָה זוֹ פַּעַם יְחִידָה שֶׁאִמָּא הָלְכָה לְבֵית כְּנֶסֶת .
