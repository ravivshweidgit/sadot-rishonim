וְכִי הָיְתָה לִי בְּרֵירָה ? וְהֵן אִישׁ עֲדַיִן לֹא מֵת מִכַּמָּה יְמֵי
מַאֲסָר וִיהְיוּ אֲרֻכִּים כְּכָל שֶׁיִּהְיוּ, וּמַה גַּם שֶׁבְּכָל זֹאת לֹא
יָשַׁבְנוּ בְּג'וּאָה ג'וּאָה.
    כָּאן עָלַי לְצַיֵּן שֶׁיּוֹתֵר מִשֶּׁשָּׁמר עָלֵינוּ הָעַרְבִי, שָׁמְרוּ הַנָּשִׁים
עָלַי לְבַל אֶסְתַּלֵּק לִי וְאֶבְרַח הַבַּיְתָה. מִשּׁוּם מַה סְּבוּרוֹת הָיוּ
שֶׁאוֹתִי בְּוַדַּאי יָבֹאוּ לְשַׁחְרֵר וּמִמֵּילָא גַּם אוֹתָן וְאִלּוּ לָהֶן
לְבַדָּן לֹא יִהְיֶה דּוֹאֵג.

    יוֹם אֶחָד בָּא עִמָּנוּאֵל, מַקְלוֹ עַל כְּתֵפוֹ וּבִקְצֵהוּ מִטַּלְטֵל
צְרֹר עָטוּף בְּמַטְלִית לְבָנָה - אֹכֶל מִן הַבַּיִת. אֵיזוֹ הַפְתָּעָה
נְעִימָה ! לְשִׂמְחָתִי לֹא הָיָה גְּבוּל אֲבָל כְּשֶׁהֻצְרַךְ לַחֲזֹר הַ־
בַּיְתָה - כִּמְעַט שֶׁהִתְחַלְתִּי לִבְכּוֹת. אָמַרְתִּי לַנָּשִׁים: "אצֵא
לְלַוּוֹתוֹ וּמִיָּד אֶחְזֹר". הֵן הִסְכִּימוּ, בְּלִית בְּרֵירָה, אֶלָּא שֶׁהִזְ־
הִירוּנִי : לַחֲזֹר מִיָּד, לֹא לְהַרְחִיק לֶכֶת, וְ־"רק שֶׁבַּעַל הַמָּלוֹן
לֹא יַרְגִּישׁ בְּצֵאתִי" וְעוֹד כהנָה וְכהנָה אֲזְהָרוֹת. חָמַקְתִּי אֵיכִי
שֶׁהוּ הַחוּצָה וְיָצָאתִי עִם עִמָּנוּאֵל אֶל מִחוּץ לָעֲיָרָה. מֵרָחוֹק
הִשְׁקַפְתִּי עַל מְטֻלָּה, כָּל־כָּךְ פָּשׁוּט וְקַל לָרוּץ הַבַּיְתָה וּבְוַדַּאי
גַּם לֹא הָיוּ דּוֹלְקִים אַחֲרַי לְתָפְשֵׂנִי, אֲבָל... הֲלֹא הִבְטַחְתִּי לַחְ־
זֹר. כְּשֶׁחָזַרְתִּי מָצָאתִי אֶת הַנָּשִׁים בְּעִצוּמוֹ שֶׁל "סִימְפּוֹזְיוֹן"
עַל אֵיךְ הִסְכִּימוּ שֶׁאֵצֵא וּמִי מֵהֶן תֵּלֵךְ לְהוֹדִיעַ עַל בְּרִיחָתִי.

    וּבְאַחַד הַיָּמִים, בַּשְּׁמִינִי לְמַאֲסָר אַחַר הַצָּהֳרַיִם, בָּא הַ־
מֻכְתָּר בִּכְבוֹדוֹ וּבְעַצְמוֹ וְהוֹדִיעַ לָנוּ עַל שִׁחְרוּרֵנוּ, מִן הֹן לְהֹן
(מִכָּאן לְכָאן), עַד שֶׁסֻּדַּר "רִשְׁמִית" עִנְיַן הַשִּׁחְרוּר - וְאוּלַי
הָיוּ לוֹ לַמֻּכְתָּר גַּם עִסּוּקִים מִשֶּׁלוֹ אוֹ בִּשְׁלִיחוּת אִשְׁתּוֹ, אַחֲרֵי
הַכֹּל לֹא בְּכָל יוֹם מִזְדַּמְּנִים "הָעִירָה" - רַד הָעֶרֶב כְּשֶׁהִתְ־
חַלְנוּ גּוֹלְשִׁים מִגִּבְעַת הָעֲיָרָה גִ'דֵידָה בְּדַרְכֵּנוּ הַבַּיְתָה. וְכָמוֹ
בְּכָל שָׁנָה וּבְכָל מָקוֹם, גַּם אוֹתוֹ עֶרֶב שֶׁל הָרֶבַע הָאַחֲרוֹן
