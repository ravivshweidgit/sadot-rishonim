שְׁלוּלִית אוֹ מִתּוֹכָהּ. מֵאָז וּמִתָּמִיד אָהַבְתִּי בַּעֲלֵי חַיִּים לְמִינֵיהֶם
וְלֹא בָחַלְתִּי בְּכָל חַי שֶׁהוּא, אִם הָיָה זֶה עַכְבָּר, צְפַרְדֵּעַ אוֹ
סְתָם חִפּוּשִׁית, כֻּלָּם כְּשֵׁרִים הָיוּ לְהִסְתַּכְּלוּת וּלְמִשְׂחָק.

    לִימִים, כְּשֶׁגָּדַלְתִּי קְצָת, הָיִיתִי מַרְבָּה לְטַיֵּל בַּגֹּרֶן רְכוּבָה
עַל הָאָתוֹן. מְאַכפת הָיִיתִי אֶת הָאָתוֹן בְּאוֹכַף הָרְכִיבָה הַקָּטָן
בַּעַל הָרְכּוּבוֹת — זֶה הַמְשַׁמֵּשׁ אֶת אַבָּא בִּמְסָעָיו לְרֹאשׁ־פִּנָּה
וְלֹא בָזֶה הַגָּדוֹל וְהָרָחָב הַמְּיֻעָד לְהוֹבֶלֶת מַיִם מִן הַמַּעְיָן —
וְהָיִיתִי דּוֹהֶרֶת עַל אֲתוֹנִי, מַעֲשֵׂה "פָּרָשׁ", הָלוֹךְ וְשׁוֹב לְאֹרְכָּהּ
שֶׁל הַגֹּרֶן. לְעִתִּים נִגְלְתָה לִי פִּטְרִיָּה מְצִיצָה מִתַּחַת לִתְלוּלִית
קַשׁ שֶׁנִּשְׁתַּיְּרָה מֵהַקַּיִץ, וְאָז הָיִיתִי מְאיטה אֶת קֶצֶב הַדְּהִירָה
וּבוֹלֶשֶׁת, לֹא תָּמִיד בְּהַצְלָחָה, אַחַר פִּטְרִיּוֹת.
כָּל זֶה הָיָה, כָּאָמוּר, בִּימֵי הַיַּלְדוּת. אוּלָם מִשֶּׁגָּדַלְנוּ קִבְּלָה
גַּם הַגֹּרֶן מַשְׁמָעוּת אַחֶרֶת לְגַבֵּינוּ, וְכָאִלּוּ גָּדְלָה עִמָּנוּ יַחַד.
שׁוּב לֹא מָצָאנוּ עִנְיָן בִּשְׁלוּלִיּוֹת הַמַּיִם וּבְסְפִיחֵי הַיְּבוּל הַיְּרַק־
רַקִּים שֶׁל הַגֹּרֶן הַחֹרְפִית, וְעִם רֵאשִׁית הַקַּיִץ מְשֶׁהוּחַל "אִיכ-
לוסָה" בִּגְדִישֵׁי הַקִּטְנִיּוֹת, בְּעַרְמוֹת שְׂעוֹרִים, שִׁבֹּלֶת־שׁוּעָל,
חִטִּים וְחומְצָה — לֹא הָיוּ לָנוּ יָמִים טוֹבִים כִּימֵי הַקַּיִץ
הָאֲרֻכִּים, תַּרְתֵּי־מַשְׁמַע, שֶׁנִּמְשְׁכוּ כִּמְעַט עַד בּוֹא הַגְּשָׁמִים.
-
מַרְבִּים לְסַפֵּר בִּלְיְלוֹת הַקַּיִץ עַל הַגֹּרֶן וַאֲנִי אָהַבְתִּי דַּוְקָא
אֶת יְמֵי הַקַּיִץ עַל הַגֹּרֶן. אֶת הָעֲבוֹדוֹת לְמִינֵיהֶן. הֵחֵל
בְּהַעֲמָסַת הַתְּבוּאָה בַּשָּׂדֶה עַל עֲגָלַת סֻלָּמוֹת, כְּשֶׁמַּכְפִּילִים
וּמְשַׁלְּשִׁים אֶת גָּבְהָהּ וְרָחְבָּהּ בַּעֲמִירֵי הַתְּבוּאָה וּמוֹבִילִים
הַגֹּרְנָה, וְאָז, כְּשֶׁכָּל הַיְּבוּל, גְּדִישִׁים גְּדִישִׁים גְּדוֹלִים מְאֹד
וּגְדוֹלִים פָּחוֹת, מְמַלְאִים אֶת הַגֹּרֶן לְמִן הַקָּצֶה וְעַד הַקָּצֶה —
מַתְחִילָה מְלֶאכֶת הַדישׁ. וְכִי יֵשׁ תַּעֲנוּג גָּדוֹל מֵאֲשֶׁר לַעֲמוֹד
