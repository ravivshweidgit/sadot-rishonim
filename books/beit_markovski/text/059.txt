הוּצַע לְאִמָּא עַ"י ח.מ. קְלוּוֶרִיסְקִי לַעֲקֹר , לְחָדְשֵׁי הַלִּמּוּ־
דִים , תֵּל־אָבִיבָה אִתָּנוּ הַיְּלָדִים וְעִם בְּנוֹתָיו שֶׁלּוֹ , לְהֶמְשֵׁךְ
הַלִּמּוּדִים בְּגִמְנַסְיָה הֶרְצְלִיָּה . אִמָּא קִבְּלָה בְּרָצוֹן אֶת הַהַצָּעָה
אֲבָל אַבָּא הִתְנַגֵּד לְכָךְ בְּכָל תֹּקֶף בִּטְעָנָה שֶׁהָעִיר תַּרְחִיק אוֹ־
תָנוּ מֵעֲבוֹדַת הָאֲדָמָה ( אוּלַי צָדַק בְּכָךְ , שֶׁכֵּן תִּרְצָה וְאֶזְרַח
שֶׁגָּדְלוּ וְלָמְדוּ בִּירוּשָׁלַיִם , לֹא פָּנוּ לְחַקְלָאוּת ).
    אֲבָל פטּוּר בְּלֹא כְלוּם אִי אֶפְשָׁר . הוּחְלַט , אֵיפֹא , עַל הֶמְשֵׁךְ
הַלִּמּוּדִים בַּבַּיִת . שֻׁלְחָן גָּדוֹל הוּצַב בְּמֶרְכָּזוֹ שֶׁל חַדְרֵנוּ , חֶדֶר
הַיְּלָדִים , מֵעָלָיו תְּלוּיָה וּמִשְׁתַּלְשֶׁלֶת מִן הַתִּקְרָה מְנוֹרַת נֶפְט -
שֶׁכֵּן הַלִּמּוּדִים בְּחֶלְקָם הִתְנַהֲלוּ בִּשְׁעוֹת הָעֶרֶב , בַּיּוֹם הָיָה
עָלֵינוּ לִלְמֹד גַּם עֲבוֹדַת הָאֲדָמָה , כֵּיצַד חַיָּב אָדָם לְהוֹצִיא
לֶחֶם מִן הָאָרֶץ .

כָּרָגִיל , נִגַּשׁ אַבָּא יָשָׁר לִפְעֻלָּה , עוֹד בְּאוֹתוֹ חֹרֶף נָתַן עֵינוֹ
בִּשְׁטַח אֲדְמַת בּוּר , לֹא גָּדוֹל וְלִלְא בְּעָלִים , זָרַע בּוֹ ( בִּשְׂכִירוּת )
שְׂעוֹרָה וּבְבֹא עֵת הַקָּצִיר - שָׂכַר פּוֹעֵל מִגְּרֵי הַצֶּדֶק שֶׁהָיוּ
בַּמּוֹשָׁבָה שֶׁקָּצַר אֶת הַשְּׂעוֹרָה בְּחֶרְמֵשׁ . יְהוּדָה וַאֲנִי , מַגובֵי
יָד עֲשׂוּיִים עֵץ תּוֹצֶרֶת בַּיִת בְּיָדֵינוּ , גּוֹבַבְנוּ אֶת הַשְּׂעוֹרָה הַפְּזוּ־
רָה בַּשָּׂדֶה שֶׁהָיָה זָרוּעַ אֲבָנִים וְרִגְבֵי עָפָר וּבְשֶׁל כָּךְ קֻצְּצוּ
מַרְבִּית הַשִּׁבֳּלִים מִן הַקַּשׁ וְנִשְׁאֲרוּ טְמוּנוֹת בֵּין הָרְגָבִים .
בְּקִצּוּר , הָיְתָה זוֹ "חַקְלָאוּת מִתְקַדֶּמֶת" בְּיוֹתֵר . אֲבָל אֵין דָּבָר.
הַנִּסָּיוֹן הִצְלִיחַ . וְהָא רְאָיָה - לַשָּׁנָה הַשְּׁנִיָּה קִבַּלְנוּ מִכְסַת
קַרְקַע כְּשֶׁל כָּל אִכָּרֵי הַמּוֹשָׁבָה וְהָפַכְנוּ לִהְיוֹת חַקְלָאִים מִן
הַמִּנְיָן .
     מִכְסַת הַקַּרְקַע כְּ־ 300 דּוּנָם , הָיְתָה מְפֻצֶּלֶת לְכַמָּה וְכַמָּה
חֶלְקוֹת בִּשְׁטָחִים , בְּכִוּוּנִים וּבִשְׁמוֹת שׁוֹנִים . הָיָה שֶׁטַח הַמַּרְג' .
מַרְג'־עיּוּן שֶׁכָּעֵת רַק מִסְתַּכְּלִים עָלָיו מֵרָחוֹק . בְּחֶלְקוֹ -
