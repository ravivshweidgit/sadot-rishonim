עֲצֵי הַזַּיִת, גַּם הֵם הֵנִיבוּ פִּרְיָם בְּשֶׁפַע וְלֹא חָסַרְנוּ זֵיתִים
טוֹבִים כָּל הַשָּׁנָה כֻּלָּהּ.
גַּם מַטַּע שְׁקֵדִים הָיָה לָּנוּ. נָטַעְנוּ אוֹתוֹ בְּחֶלְקוֹ הַמִּישֹׁרִי
הַמְּרֻחָק מֵהֶחָצֵר, שֶׁל הַמִּגְרָשׁ. אֶת הָאֲבָנִים הַזְּרוּעוֹת בַּשֶּׁטַח,
סִקַּלְנוּ וְגָדַרְנוּ בָּהֶם, כַּמְּקֻבָּל, אֶת הַשֶּׁטַח כֻּלּוֹ.
הַמַּטָּע — כְּ־10 דּוּנָם — הָיָה בְּאוֹתָהּ תְּקוּפָה וּבְאוֹתָם
תְּנָאִים, עָנָף מִשְׁקִי בַּעַל חֲשִׁיבוּת. הַשְּׁקֵדִים — מֵהַזַּן "פְּרִינִי
צִיסָה" שֶׁקְּלִיפָתוֹ רַכָּה וְתוֹכוֹ מָתוֹק וְטָעִים.
   גַּן הַשְּׁקֵדִים שֶׁלָּנוּ הָיָה מְעֻבָּד וּמְטֻפָּח, אֲבָל הַיְּבוּל לֹא
הָיָה רַב. הַפְּרִי בְּחֶלְקוֹ, נִקְטַף וְנֶאֱכַל, עְיֵי כָּל עוֹבֵר וָשָׁב,
בְּעוֹדוֹ יָרֹק וְרַךְ וְלִכְלַל הַבְשָׁלָה הִגִּיעַ רַק חֵלֶק קָטָן מֵהַפֵּרוֹת.
עִם צֵאתֵנוּ אֶת מְטֻלָּה, נוֹסַף גַּם מַטַּע הַשְּׁקֵדִים אֶל כָּל
הַנִּכְסֵי־דְּלָא־נִיְידֵי שֶׁטִּפַּחְנוּ וְהִשְׁאַרְנוּ לְלֹא תְּמוּרָה.
   וְאִם אֶת שֶׁטַח הַמַּטָּע הָיָה עָלֵינוּ "רַק לְסַקֵּל" מֵאֲבָנָיו —
הָיוּ בַּגִּנָּה שֶׁלִּפְנֵי הַבַּיִת גַּם סְלָעִים וְהֻצְרַכְנוּ לְעָקְרָם בְּמוֹט
בַּרְזֶל. הַמּוֹט הָיָה עָבֶה וְכָבֵד, מְחֻודד בְּקָצֵהוּ הָאֶחָד וְשָׁטוּחַ
בְּקָצֵה הָאַחֵר; הִשְׁתַּמַּשְׁנוּ בִּשְׁנֵי הַקְּצָווֹת חֲלִיפוֹת לְפִי הַצֹּרֶךְ.
גַּם מוֹט דַּק וְקַל דְּמוּי אֵת חֲפִירָה קָטָן הָיָה לָּנוּ וּבוֹ עָקַרְנוּ
סְלָעִים לֹא גְּדוֹלִים וְלֹא עֲמֻקִּים בִּמְיֻחָד. כֵּן שִׁמֵּשׁ לָנוּ
הַמּוֹט הַקָּטָן לְגִלּוּי הַסֶּלַע לְעָמְקוֹ, מִתַּחַת פְּנֵי הַקַּרְקַע.
אָהַבְתִּי לְהַכּוֹת בַּמּוֹט וְלְ"פצֵח" אֶת הַסֶּלַע וְהָיִיתִי מְלֵאַת
גַּאֲוָה כְּשֶׁהִצְלַחְתִּי לְעָקְרוֹ מִן הַשֹּׁרֶשׁ.
   בַּגִּנָּה נָטַעְנוּ שְׁנֵי עֲצֵי תוּת וְזֹאת — כִּמְזוֹן לְתוֹלְעֵי הַמֶּשִׁי.
הֲלֹא גַּם הֵם בגדַר "גִּדּוּל" שֶׁיֵּשׁ לְהִתְנַסּוֹת בּוֹ; וְאָמְנָם הִתְ־
נַסִּינוּ. כַּמָּה עֶשְׂרוֹת (וְאוּלַי מֵאוֹת) פִּקָּעוֹת שֶׁל תּוֹלַעַת הַמֶּשִׁי
הָיוּ לָנוּ וַאֲנַחְנוּ גִּדַּלְנוּ וְטִפַּחְנוּ אוֹתָם עַד אֲשֶׁר... אֵינֶנִּי זוֹכֶרֶת
מֶה עָלָה בְּגוֹרָלָם, אֲבָל בִּגְדֵי מֶשִׁי לֹא הָיוּ לָנוּ; בְּזֹאת אֲנִי
בְּטוּחָה.
